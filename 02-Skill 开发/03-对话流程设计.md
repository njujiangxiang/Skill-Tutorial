# 03 - 对话流程设计

## 🎯 学习目标

- 理解对话流程的重要性
- 掌握对话状态管理
- 学会设计多轮对话
- 实现上下文理解

## 💬 对话类型

### 单轮对话

```
用户：北京天气怎么样？
AI:  北京今天晴，25°C。
```

**特点**:
- 一次请求，一次响应
- 无需记忆上下文
- 实现简单

### 多轮对话

```
用户：北京天气怎么样？
AI:  北京今天晴，25°C。

用户：那上海呢？
AI:  上海今天小雨，18°C。

用户：明天呢？
AI:  上海明天多云，20°C。
```

**特点**:
- 需要记忆上下文
- 支持省略和指代
- 实现复杂

## 🔄 对话状态管理

### 状态机模型

```javascript
class ConversationState {
  constructor() {
    this.currentIntent = null;
    this.slots = {};
    this.history = [];
  }

  // 更新意图
  setIntent(intent) {
    this.currentIntent = intent;
  }

  // 填充槽位
  fillSlot(name, value) {
    this.slots[name] = value;
  }

  // 检查是否所有槽位都已填充
  isComplete() {
    const requiredSlots = this.getRequiredSlots();
    return requiredSlots.every(slot => this.slots[slot]);
  }

  // 获取缺失的槽位
  getMissingSlots() {
    const requiredSlots = this.getRequiredSlots();
    return requiredSlots.filter(slot => !this.slots[slot]);
  }

  // 重置状态
  reset() {
    this.currentIntent = null;
    this.slots = {};
  }
}
```

### 上下文管理

```javascript
class ContextManager {
  constructor(maxHistory = 10) {
    this.contexts = new Map();
    this.maxHistory = maxHistory;
  }

  // 设置上下文
  set(userId, key, value, ttl = 300000) {
    const userContext = this.contexts.get(userId) || {};
    userContext[key] = {
      value,
      expires: Date.now() + ttl,
    };
    this.contexts.set(userId, userContext);
  }

  // 获取上下文
  get(userId, key) {
    const userContext = this.contexts.get(userId);
    if (!userContext) return null;

    const item = userContext[key];
    if (!item) return null;

    // 检查是否过期
    if (Date.now() > item.expires) {
      delete userContext[key];
      return null;
    }

    return item.value;
  }

  // 添加到历史
  addToHistory(userId, message) {
    const history = this.contexts.get(userId)?.history || [];
    history.push({
      ...message,
      timestamp: Date.now(),
    });

    // 保持历史记录数量
    if (history.length > this.maxHistory) {
      history.shift();
    }

    this.set(userId, 'history', history);
  }

  // 清除上下文
  clear(userId) {
    this.contexts.delete(userId);
  }
}
```

## 📝 对话流程设计

### 槽位填充模式

```javascript
class SlotFillingDialog {
  constructor() {
    this.slots = {
      city: {
        prompt: '请问您想查询哪个城市的天气？',
        validate: (value) => value.length > 0,
      },
      date: {
        prompt: '请问您想查询哪天的天气？（今天/明天/后天）',
        validate: (value) => ['今天', '明天', '后天'].includes(value),
        default: '今天',
      },
    };
  }

  async process(input, context) {
    // 提取槽位值
    for (const [slotName, slotConfig] of Object.entries(this.slots)) {
      if (!context.slots[slotName]) {
        const value = this.extractSlot(input, slotName);
        
        if (value) {
          context.slots[slotName] = value;
        } else {
          // 请求缺失的槽位
          return {
            type: 'question',
            message: slotConfig.prompt,
          };
        }
      }
    }

    // 所有槽位已填充，执行查询
    return {
      type: 'answer',
      data: await this.execute(context.slots),
    };
  }

  extractSlot(input, slotName) {
    // 简单的提取逻辑
    if (slotName === 'city') {
      const match = input.match(/(.*) 天气/);
      if (match) return match[1];
    }
    
    if (slotName === 'date') {
      for (const date of ['今天', '明天', '后天']) {
        if (input.includes(date)) return date;
      }
    }

    return null;
  }
}
```

### 对话流程示例

```javascript
class WeatherDialog {
  constructor() {
    this.state = new ConversationState();
    this.context = new ContextManager();
  }

  async process(userId, input) {
    // 添加到历史
    this.context.addToHistory(userId, { role: 'user', content: input });

    // 检查是否有上下文引用
    input = this.resolveReferences(input, userId);

    // 意图识别
    const intent = this.recognizeIntent(input);
    
    if (!intent) {
      return {
        message: '抱歉，我没有理解您的意思。您想查询天气吗？',
      };
    }

    // 更新状态
    this.state.setIntent(intent);

    // 槽位填充
    const extraction = this.extractSlots(input, intent);
    Object.assign(this.state.slots, extraction);

    // 检查是否所有必需槽位都已填充
    if (!this.state.isComplete()) {
      const missingSlot = this.state.getMissingSlots()[0];
      return {
        message: this.getPrompt(missingSlot),
        type: 'question',
      };
    }

    // 执行查询
    const result = await this.executeQuery(this.state.slots);

    // 格式化结果
    const response = this.formatResponse(result);

    // 添加到历史
    this.context.addToHistory(userId, { role: 'assistant', content: response });

    // 重置状态
    this.state.reset();

    return {
      message: response,
      type: 'answer',
    };
  }

  resolveReferences(input, userId) {
    // 处理"那上海呢？"中的指代
    if (input.includes('那') && input.includes('呢')) {
      const lastCity = this.context.get(userId, 'lastCity');
      if (lastCity) {
        return input.replace('那', lastCity).replace('呢', '');
      }
    }
    return input;
  }

  recognizeIntent(input) {
    if (/天气 | 气温 | 下雨 | 温度/.test(input)) {
      return 'weather_query';
    }
    return null;
  }

  extractSlots(input, intent) {
    const slots = {};

    if (intent === 'weather_query') {
      // 提取城市
      const cityMatch = input.match(/(.*) 天气/);
      if (cityMatch) {
        slots.city = cityMatch[1].trim();
      }

      // 提取日期
      for (const date of ['今天', '明天', '后天']) {
        if (input.includes(date)) {
          slots.date = date;
        }
      }
    }

    return slots;
  }

  getPrompt(slotName) {
    const prompts = {
      city: '请问您想查询哪个城市的天气？',
      date: '请问您想查询哪天的天气？（今天/明天/后天，默认为今天）',
    };
    return prompts[slotName] || '请提供更多信息';
  }

  async executeQuery(slots) {
    // 调用天气 API
    const city = slots.city || '北京';
    const date = slots.date || '今天';
    
    // 模拟 API 调用
    return {
      city,
      date,
      temperature: 25,
      condition: '晴',
    };
  }

  formatResponse(result) {
    return `${result.city}${result.date}${result.condition}，气温${result.temperature}°C。`;
  }
}
```

class ProactiveDialog {
  async process(input, context) {
    const slots = this.extractSlots(input);

    if (!slots.city) {
      return {
        message: '您想查询哪个城市的天气呢？北京、上海、还是广州？',
        suggestions: ['北京', '上海', '广州', '深圳'],

    if (!slots.date) {
      return {
        message: `好的，${slots.city}的天气。您想查询今天还是明天的？`,
        suggestions: ['今天', '明天', '后天'],

    }

    // 执行查询
    return { message: '...' };
  }
}

### 2. 确认机制

```javascript
class ConfirmDialog {
  async process(input, context) {
    const slots = this.extractSlots(input);

    // 如果是确认
    if (['是的', '对', '确认'].includes(input.trim())) {
      return await this.execute(context.pendingSlots);
    }

    // 如果是否认
    if (['不是', '不对', '取消'].includes(input.trim())) {
      return { message: '好的，已取消。' };
    }

    // 存储待确认的槽位
    context.pendingSlots = slots;

    // 请求确认
    const summary = this.summarize(slots);
    return {
      message: `我理解您想查询${summary}，对吗？`,
      type: 'confirmation',
    };
  }

  summarize(slots) {
    return `${slots.city}${slots.date || '今天'}的天气`;
  }
}
```

### 3. 纠错处理

```javascript
class ErrorRecoveryDialog {
  async process(input, context) {
    try {
      return await this.normalProcess(input, context);
    } catch (error) {
      // 记录错误
      logger.error('对话处理失败', error);

      // 提供友好的错误消息
      if (error.code === 'CITY_NOT_FOUND') {
        return {
          message: `抱歉，没有找到"${error.city}"的天气信息。您是不是想查询其他城市？`,
          suggestions: ['北京', '上海', '广州'],
        };
      }

      if (error.code === 'API_ERROR') {
        return {
          message: '抱歉，天气服务暂时不可用，请稍后再试。',
        };
      }

      // 通用错误
      return {
        message: '抱歉，我没有理解。您能再说一遍吗？',
      };
    }
  }
}
```

## 📊 对话评估

### 指标定义

```javascript
const metrics = {
  // 任务完成率
  completionRate: (completed, total) => completed / total,

  // 平均对话轮数
  averageTurns: (turns) => turns.reduce((a, b) => a + b, 0) / turns.length,

  // 用户满意度
  satisfaction: (ratings) => ratings.reduce((a, b) => a + b, 0) / ratings.length,

  // 错误率
  errorRate: (errors, total) => errors / total,
};
```

### 日志记录

```javascript
class DialogLogger {
  log(userId, dialog) {
    const log = {
      userId,
      timestamp: Date.now(),
      input: dialog.input,
      output: dialog.output,
      intent: dialog.intent,
      slots: dialog.slots,
      duration: dialog.duration,
      success: dialog.success,
    };

    // 保存到日志文件
    fs.appendFileSync('dialog.log', JSON.stringify(log) + '\n');
  }

  // 分析日志
  analyze() {
    const logs = fs.readFileSync('dialog.log', 'utf-8')
      .split('\n')
      .filter(line => line)
      .map(line => JSON.parse(line));

    return {
      totalDialogs: logs.length,
      successRate: logs.filter(l => l.success).length / logs.length,
      averageDuration: logs.reduce((sum, l) => sum + l.duration, 0) / logs.length,
    };
  }
}
```
