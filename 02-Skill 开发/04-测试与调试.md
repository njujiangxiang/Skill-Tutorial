### 测试工具函数

```javascript
// __tests__/weather.test.js
import { getWeather } from '../src/weather.js';

// Mock fetch
global.fetch = jest.fn();

describe('getWeather', () => {
  beforeEach(() => {
    fetch.mockClear();
  });

  test('应该返回天气数据', async () => {
    // 准备测试数据
    const mockData = {
      location: { name: '北京', country: '中国' },
      current: { temp_c: 25, condition: { text: '晴' }, humidity: 60 },
    };

    fetch.mockResolvedValueOnce({
      ok: true,
      json: async () => mockData,
    });

    // 执行测试
    const result = await getWeather({ city: '北京' });

    // 断言
    expect(result.success).toBe(true);
    expect(result.data.city).toBe('北京');
    expect(result.data.temperature).toBe(25);
    expect(result.data.condition).toBe('晴');
  });

  test('应该处理 API 错误', async () => {
    fetch.mockResolvedValueOnce({
      ok: false,
      status: 404,
    });

    const result = await getWeather({ city: '北京' });

    expect(result.success).toBe(false);
    expect(result.error.code).toBe('WEATHER_API_ERROR');
  });

  test('应该验证参数', async () => {
    await expect(getWeather({}))
      .rejects
      .toThrow('城市名不能为空');

    await expect(getWeather({ city: 123 }))
      .rejects
      .toThrow('城市名必须是字符串');
  });

  test('应该使用缓存', async () => {
    const mockData = {
      location: { name: '北京' },
      current: { temp_c: 25 },
    };

    fetch.mockResolvedValueOnce({
      ok: true,
      json: async () => mockData,
    });

    // 第一次调用
    const result1 = await getWeather({ city: '北京' });
    expect(fetch).toHaveBeenCalledTimes(1);

    // 第二次调用（应该使用缓存）
    const result2 = await getWeather({ city: '北京' });
    expect(fetch).toHaveBeenCalledTimes(1); // 仍然是 1 次

    expect(result1.data.temperature).toBe(result2.data.temperature);
  });
});
```

## 🔍 集成测试

### 端到端测试

```javascript
// __tests__/integration.test.js
import { createTodo, listTodos, completeTodo } from '../index.js';

describe('待办事项集成测试', () => {
  test('完整的任务流程', async () => {
    // 1. 创建任务
    const createResult = await createTodo({
      title: '集成测试任务',
      priority: 'high',
    });

    expect(createResult.success).toBe(true);
    const todoId = createResult.data.todo.id;

    // 2. 查看任务列表
    const listResult = await listTodos({ status: 'pending' });
    expect(listResult.success).toBe(true);
    expect(listResult.data.todos.some(t => t.id === todoId)).toBe(true);

    // 3. 完成任务
    const completeResult = await completeTodo({ id: todoId });
    expect(completeResult.success).toBe(true);

    // 4. 验证任务状态
    const listResult2 = await listTodos({ status: 'completed' });
    expect(listResult2.data.todos.some(t => t.id === todoId)).toBe(true);

    // 5. 清理
    await storage.delete(todoId);
  });
});
```

### 日志调试

```javascript
// src/logger.js
import fs from 'fs';
import path from 'path';

const LOG_DIR = path.join(process.env.HOME || '.', '.openclaw', 'logs');
const LOG_FILE = path.join(LOG_DIR, 'skill.log');

// 确保日志目录存在
fs.mkdirSync(LOG_DIR, { recursive: true });

export const logger = {
  debug(message, data) {
    this.write('DEBUG', message, data);
  },

  info(message, data) {
    this.write('INFO', message, data);
  },

  warn(message, data) {
    this.write('WARN', message, data);
  },

  error(message, data) {
    this.write('ERROR', message, data);
  },

  write(level, message, data) {
    const log = {
      timestamp: new Date().toISOString(),
      level,
      message,
      data,
    };

    fs.appendFileSync(LOG_FILE, JSON.stringify(log) + '\n');

    // 开发环境下输出到控制台
    if (process.env.NODE_ENV === 'development') {
      console.log(`[${level}] ${message}`, data || '');
    }
  },
};

// 使用示例
async function getWeather(city) {
  logger.debug('开始查询天气', { city });

  try {
    const result = await api.call(city);
    logger.info('天气查询成功', { city, result });
    return result;
  } catch (error) {
    logger.error('天气查询失败', { city, error: error.message });
    throw error;
  }
}
```

### 性能分析

```javascript
// src/profiler.js
export class Profiler {
  constructor() {
    this.timers = new Map();
  }

  start(name) {
    this.timers.set(name, Date.now());
  }

  end(name) {
    const start = this.timers.get(name);
    if (!start) return;

    const duration = Date.now() - start;
    console.log(`[${name}] 耗时：${duration}ms`);
    this.timers.delete(name);

    return duration;
  }

  async measure(name, fn) {
    this.start(name);
    try {
      return await fn();
    } finally {
      this.end(name);
    }
  }
}

// 使用示例
const profiler = new Profiler();

async function getWeather(city) {
  return profiler.measure('getWeather', async () => {
    const response = await fetch(apiUrl);
    const data = await response.json();
    return formatWeather(data);
  });
}
```

```javascript
// ❌ 低覆盖率的代码
function process(input) {
  if (input) {
    return input.toUpperCase();
  }
}

// ✅ 高覆盖率的代码
function process(input) {
  // 处理 null/undefined
  if (input === null || input === undefined) {
    return '';
  }

  // 处理非字符串
  if (typeof input !== 'string') {
    throw new Error('输入必须是字符串');
  }

  // 处理空字符串
  if (input.trim() === '') {
    return '';
  }

  return input.trim().toUpperCase();
}

// 完整的测试
describe('process', () => {
  test('应该转换大写', () => {
    expect(process('hello')).toBe('HELLO');
  });

  test('应该处理 null', () => {
    expect(process(null)).toBe('');
  });

  test('应该处理 undefined', () => {
    expect(process(undefined)).toBe('');
  });

  test('应该处理非字符串', () => {
    expect(() => process(123)).toThrow('输入必须是字符串');
  });

  test('应该处理空字符串', () => {
    expect(process('   ')).toBe('');
  });
});
```
