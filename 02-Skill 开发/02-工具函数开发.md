## ✅ 参数验证

### 基础验证

```javascript
function validateParams(params) {
  // 检查必填参数
  if (!params.city) {
    throw new Error('缺少必填参数：city');
  }
  
  // 检查类型
  if (typeof params.city !== 'string') {
    throw new Error('参数 city 必须是字符串');
  }
  
  // 检查长度
  if (params.city.length > 100) {
    throw new Error('参数 city 过长');
  }
}
```

### 高级验证

```javascript
const validators = {
  // 字符串验证
  string: (value, options = {}) => {
    if (typeof value !== 'string') {
      throw new Error('必须是字符串');
    }
    if (options.minLength && value.length < options.minLength) {
      throw new Error(`长度不能小于${options.minLength}`);
    }
    if (options.maxLength && value.length > options.maxLength) {
      throw new Error(`长度不能大于${options.maxLength}`);
    }
  },
  
  // 数字验证
  number: (value, options = {}) => {
    if (typeof value !== 'number') {
      throw new Error('必须是数字');
    }
    if (options.min !== undefined && value < options.min) {
      throw new Error(`不能小于${options.min}`);
    }
    if (options.max !== undefined && value > options.max) {
      throw new Error(`不能大于${options.max}`);
    }
  },
  
  // 枚举验证
  enum: (value, allowedValues) => {
    if (!allowedValues.includes(value)) {
      throw new Error(`必须是以下值之一：${allowedValues.join(', ')}`);
    }
  },
};

// 使用示例
function validateParams(params) {
  validators.string(params.city, { minLength: 1, maxLength: 100 });
  validators.number(params.days, { min: 1, max: 10 });
  validators.enum(params.unit, ['celsius', 'fahrenheit']);
}
```

## 🐛 错误处理

### 基础错误处理

```javascript
async function getWeather(params) {
  try {
    const response = await fetch(apiUrl);
    
    if (!response.ok) {
      throw new Error(`API 错误：${response.status}`);
    }
    
    const data = await response.json();
    
    return {
      success: true,
      data,
    };
  } catch (error) {
    return {
      success: false,
      error: {
        code: 'WEATHER_API_ERROR',
        message: error.message,
      },
    };
  }
}
```

### 自定义错误类

```javascript
class SkillError extends Error {
  constructor(code, message, statusCode = 500) {
    super(message);
    this.name = 'SkillError';
    this.code = code;
    this.statusCode = statusCode;
  }
}

class ValidationError extends SkillError {
  constructor(message) {
    super('VALIDATION_ERROR', message, 400);
    this.name = 'ValidationError';
  }
}

class ApiError extends SkillError {
  constructor(message, statusCode = 500) {
    super('API_ERROR', message, statusCode);
    this.name = 'ApiError';
  }
}

// 使用示例
async function getWeather(params) {
  try {
    // 参数验证
    if (!params.city) {
      throw new ValidationError('城市名不能为空');
    }
    
    // API 调用
    const response = await fetch(apiUrl);
    if (!response.ok) {
      throw new ApiError('天气 API 调用失败', response.status);
    }
    
    const data = await response.json();
    
    return { success: true, data };
  } catch (error) {
    logger.error('天气查询失败', error);
    throw error;
  }
}
```

class WeatherFormatter {
  formatCurrent(data) {
    return {
      location: {
        city: data.location.name,
        country: data.location.country,
      },
      current: {
        temperature: this.formatTemperature(data.current.temp_c),
        condition: data.current.condition.text,
        icon: this.getWeatherIcon(data.current.condition.code),
        humidity: data.current.humidity,
        windSpeed: data.current.wind_kph,
      },
      timestamp: new Date().toISOString(),
    };
  }
  
  formatForecast(data) {
    return {
      location: data.location.name,
      forecasts: data.forecast.forecastday.map(day => ({
        date: day.date,
        maxTemp: this.formatTemperature(day.day.maxtemp_c),
        minTemp: this.formatTemperature(day.day.mintemp_c),
        condition: day.day.condition.text,
        chanceOfRain: day.day.daily_chance_of_rain,
      })),
    };
  }
  
  formatTemperature(celsius) {
    return {
      celsius: Math.round(celsius),
      fahrenheit: Math.round(celsius * 9/5 + 32),
    };
  }
  
  getWeatherIcon(code) {
    const icons = {
      1000: '☀️',
      1003: '⛅',
      1006: '☁️',
      1063: '🌧️',
      1066: '🌨️',
    };
    return icons[code] || '🌡️';
  }
}

## 🔧 实用工具函数

### 缓存工具

```javascript
class Cache {
  constructor(ttl = 60000) {
    this.cache = new Map();
    this.ttl = ttl;
  }
  
  async get(key) {
    const item = this.cache.get(key);
    if (!item) return null;
    
    if (Date.now() > item.expiry) {
      this.cache.delete(key);
      return null;
    }
    
    return item.data;
  }
  
  async set(key, data) {
    this.cache.set(key, {
      data,
      expiry: Date.now() + this.ttl,
    });
  }
  
  async delete(key) {
    this.cache.delete(key);
  }
  
  async clear() {
    this.cache.clear();
  }
}

// 使用示例
const cache = new Cache(5 * 60 * 1000); // 5 分钟

async function getWeather(city) {
  const cacheKey = `weather:${city}`;
  
  // 尝试从缓存获取
  let data = await cache.get(cacheKey);
  if (data) {
    return { success: true, data, fromCache: true };
  }
  
  // 从 API 获取
  data = await fetchWeather(city);
  
  // 存入缓存
  await cache.set(cacheKey, data);
  
  return { success: true, data, fromCache: false };
}
```

### 重试工具

```javascript
async function retry(fn, options = {}) {
  const {
    maxRetries = 3,
    delay = 1000,
    backoff = 2,
  } = options;
  
  let lastError;
  
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn();
    } catch (error) {
      lastError = error;
      
      if (i === maxRetries - 1) {
        break;
      }
      
      const waitTime = delay * Math.pow(backoff, i);
      logger.warn(`重试 ${i + 1}/${maxRetries}`, error);
      await new Promise(resolve => setTimeout(resolve, waitTime));
    }
  }
  
  throw lastError;
}

// 使用示例
async function getWeather(city) {
  return retry(
    async () => {
      const response = await fetch(apiUrl);
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}`);
      }
      return response.json();
    },
    { maxRetries: 3, delay: 1000 }
  );
}
```
